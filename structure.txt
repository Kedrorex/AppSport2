//Чтобы обновить древо папок и файлов вставить команду в PowerSchell в корне проекта

Get-ChildItem -Recurse -Force -Exclude node_modules |
    Where-Object { $_.FullName -notmatch '\\node_modules\\' } |
    ForEach-Object {
        $_.FullName.Replace((Get-Location).Path + '\','')
    } | Out-File structure.txt -Encoding UTF8
